-- Проверяем структуру
WITH robots_content AS (
    SELECT content FROM seo_robots WHERE is_active = TRUE ORDER BY id DESC LIMIT 1
)
SELECT 
    'Yandex' as bot,
    CASE 
        WHEN content LIKE '%User-agent: Yandex%' THEN '✅ Found'
        ELSE '❌ Missing'
    END as status
FROM robots_content
UNION ALL
SELECT 
    'Googlebot',
    CASE 
        WHEN content LIKE '%User-agent: Googlebot%' THEN '✅ Found'
        ELSE '❌ Missing'
    END
FROM robots_content
UNION ALL
SELECT 
    'Default (*)',
    CASE 
        WHEN content LIKE '%User-agent: *%' THEN '✅ Found'
        ELSE '❌ Missing'
    END
FROM robots_content
UNION ALL
SELECT 
    'Sitemap',
    CASE 
        WHEN content LIKE '%Sitemap:%' THEN '✅ Found'
        ELSE '❌ Missing'
    END
FROM robots_content;
