-- ============================================
-- ПРОВЕРКА ОБНОВЛЕНИЯ
-- ============================================

-- Проверяем актуальную версию
SELECT 
    id,
    created_by,
    created_at,
    updated_at,
    LENGTH(content) as content_length,
    CASE 
        WHEN content LIKE '%sitemap.xml%' AND content LIKE '%sitemap-index.xml%' THEN '✅ Both sitemaps present'
        WHEN content LIKE '%sitemap.xml%' THEN '⚠️ Only main sitemap'
        WHEN content LIKE '%sitemap-index.xml%' THEN '⚠️ Only index sitemap'
        ELSE '❌ No sitemap found'
    END as sitemap_status,
    LEFT(content, 200) || '...' as preview
FROM seo_robots 
WHERE is_active = TRUE 
ORDER BY id DESC 
LIMIT 1;

-- ============================================
-- ВАЛИДАЦИЯ ROBOTS.TXT
-- ============================================

SELECT * FROM validate_robots_advanced(
    (SELECT content FROM seo_robots WHERE is_active = TRUE ORDER BY id DESC LIMIT 1)
)
WHERE is_valid = FALSE;

-- ============================================
-- ПРОВЕРКА НАЛИЧИЯ SITEMAP
-- ============================================

DO $$ 
DECLARE
    v_content TEXT;
    v_has_main BOOLEAN;
    v_has_index BOOLEAN;
BEGIN
    SELECT content INTO v_content 
    FROM seo_robots 
    WHERE is_active = TRUE 
    ORDER BY id DESC LIMIT 1;
    
    v_has_main := v_content LIKE '%sitemap.xml%';
    v_has_index := v_content LIKE '%sitemap-index.xml%';
    
    RAISE NOTICE '========================================';
    RAISE NOTICE '📋 SITEMAP CHECK';
    RAISE NOTICE '========================================';
    RAISE NOTICE 'Main sitemap: %', CASE WHEN v_has_main THEN '✅ Present' ELSE '❌ Missing' END;
    RAISE NOTICE 'Index sitemap: %', CASE WHEN v_has_index THEN '✅ Present' ELSE '❌ Missing' END;
    
    IF v_has_main AND v_has_index THEN
        RAISE NOTICE '✅ Both sitemaps configured correctly!';
    ELSE
        RAISE NOTICE '⚠️ Some sitemaps are missing!';
    END IF;
    RAISE NOTICE '========================================';
END $$;

-- ============================================
-- СРАВНЕНИЕ С ПРЕДЫДУЩЕЙ ВЕРСИЕЙ
-- ============================================

WITH current AS (
    SELECT content, updated_at 
    FROM seo_robots 
    WHERE is_active = TRUE 
    ORDER BY id DESC LIMIT 1
),
previous AS (
    SELECT content, updated_at 
    FROM seo_robots_history 
    ORDER BY changed_at DESC LIMIT 1
)
SELECT 
    '📊 Version comparison' as info,
    current.updated_at as current_version,
    previous.changed_at as previous_version,
    LENGTH(current.content) as current_length,
    LENGTH(previous.content) as previous_length,
    CASE 
        WHEN current.content LIKE '%sitemap-index.xml%' AND NOT previous.content LIKE '%sitemap-index.xml%' 
        THEN '🆕 Added sitemap-index.xml'
        WHEN LENGTH(current.content) > LENGTH(previous.content) THEN '📈 Larger'
        WHEN LENGTH(current.content) < LENGTH(previous.content) THEN '📉 Smaller'
        ELSE '➡️ Same size'
    END as changes
FROM current, previous;

-- ============================================
-- СОЗДАНИЕ БЭКАПА ПЕРЕД ОБНОВЛЕНИЕМ
-- ============================================

INSERT INTO seo_robots_backup (id, content, is_active, created_by, created_at, updated_at)
SELECT id, content, is_active, created_by, created_at, updated_at
FROM seo_robots 
WHERE is_active = TRUE 
ORDER BY id DESC LIMIT 1;
